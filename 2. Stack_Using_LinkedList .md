#include <bits/stdc++.h>
using namespace std;

class StackNode {
public:
    int data;
    StackNode* next;
};
 
bool isEmpty(StackNode* top_ref)
{   if(top_ref == NULL)
     return true;
    else
     return false;
}
 
void push(StackNode** top_ref, int data)
{
    StackNode* sNode = new StackNode();
    sNode->data = data;
    sNode->next = *top_ref;
    *top_ref = sNode;
}
 
void pop(StackNode** top_ref)
{
    if (isEmpty(*top_ref))
      cout<<"Stack underflow";
    StackNode* temp = *top_ref;
    *top_ref = (*top_ref)->next;
    delete temp;
    return;
}
 
void peek(StackNode* top)
{
    if (isEmpty(top))
    {    cout<<"Stack is empty";
        return;
    }
    cout<<top->data<<"\n";
    return;
}

void display(StackNode* top)
{   if (isEmpty(top))
    {    cout<<"Stack is empty";
        return;
    }
    while(top!=NULL)
    { 
        cout<<top->data<<"\n";
        top=top->next;
    }
}

int main()
{
    StackNode* top = NULL;
    int ch,val;
    cout<<"1) Push in stack"<<endl;
    cout<<"2) Pop from stack"<<endl;
    cout<<"3) Display stack"<<endl;
    cout<<"4) Top-most element"<<endl;
    cout<<"5) EXIT"<<endl;
    do {
        cout<<"Enter choice: "<<endl;
        cin>>ch;
        switch(ch) {
            case 1: {
                cout<<"Enter value to be pushed:"<<endl;
                cin>>val;
                push(&top,val);
                break;
            }
            case 2: {
                pop(&top);
                break;
            }
            case 3: {
                display(top);
                break;
            }
            case 4: {
                peek(top);
                break;
            }
            case 5: {
                cout<<"EXIT"<<endl;
                break;
            }
            default: {
                cout<<"Invalid Choice"<<endl;
            }
        }
    }while(ch!=5);
    return 0;
}
